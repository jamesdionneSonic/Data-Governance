CREATE PROCEDURE [cmag].[pDailyCashSummaryRpt]                    
(
@Company Varchar(255),
@YearMonth Varchar(7)
)
AS
BEGIN 

SET NOCOUNT ON;
SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;

/*===============================================================================                    
 Purpose:         To collect Daily Activity Summary Records for Cash Management's Daily
					Cash Summary Report.
 
 Created By:      Doug Morgan and BJ Richard on 4/19/2010

 Params:          @Company, @YearMonth
                    
 Optional:                    
                    
 Revisions:                    
                   
 Call:                  exec pDailyCashSummaryRpt 'Sonic Automotive, Inc', '2009-11'                   
                    
===============================================================================*/  


--drop table #BeginBalDay

DECLARE @myYearMonth Char(7)
DECLARE @LoopDate datetime
DECLARE @LastMonth Datetime
DECLARE @LastOfMonthBal as Money
DECLARE @Act_DLR_ID as numeric(18,0)
DECLARE @BalanceDate as Datetime 
DECLARE @BeginningBalance as Money
DECLARE @Deposit as Money
DECLARE @CDA as Money
DECLARE @DailyBalance as Money
DECLARE @Loopcount int

--Select @Company = 'Sonic Automotive, Inc'
--Select @YearMonth = '2009-11'

SELECT @Act_DLR_ID = (Select Dlr_id from CMS.cmag.tblDealershipInfo
where Dlr_dealership = @Company)  -- This will be a report parameter

SELECT @myYearMonth = @YearMonth -- this will be a report parameter.

Create Table #Days (Rowid int identity(1,1),FullDate datetime)

Insert into #Days
select Fulldate from CMS.dbo.DimDate
where  CalendarYearMonth = @myYearMonth
order by FullDate desc

select @Loopcount = @@ROWCOUNT

Create Table #DailyCashSummary (
	[DCS_Act_Dealership] varchar(255) NOT NULL,	
	[DCS_Act_Date] [datetime] NOT NULL,
	[DCS_Cash_Balance] [money] NOT NULL,
	[DCS_Deposits] [money] NULL,
	[DCS_CDA] [money] NULL)





Select @BalanceDate = (select MAX(act_date) from [CMS].[cmag].[tblActivity] 
where Act_Date < (select MIN(fulldate) from #days))

Select @BeginningBalance = (select [Act_Daily_Cash_Balance] from [CMS].[cmag].[tblActivity] 
where Act_DLR_ID = @Act_DLR_ID and Act_Date = @BalanceDate) 

Select @DailyBalance = @BeginningBalance

While @Loopcount >0
Begin
Select @LoopDate = (Select Fulldate from #Days where Rowid = @Loopcount)

Select @Deposit = (Select isnull([Act_Daily_Deposits],0.00) FROM [CMS].[cmag].[tblActivity] where Act_Date = @LoopDate
and Act_DLR_ID = @Act_DLR_ID)

Select @CDA = (Select isnull([Act_Daily_CDA],0.00) from [CMS].[cmag].[tblActivity] where Act_Date = @LoopDate
 and Act_DLR_ID = @Act_DLR_ID)

Insert into #DailyCashSummary
Select @Company,@Loopdate, @DailyBalance + isnull(@Deposit,0.00) + isnull(@CDA,0.00), isnull(@Deposit,0.00),isnull(@CDA,0.00)

Select @DailyBalance = @DailyBalance + isnull(@Deposit,0.00) + isnull(@CDA,0.00)


Select @Loopcount = @Loopcount-1
print @Company
End

Select * from #DailyCashSummary

drop table #days
drop table #dailycashsummary 

End
