CREATE  FUNCTION [cmag].[tvfn_DailyActivity] 
(	
 @ActivityDate datetime,	@User_Id VARCHAR(20)
)
RETURNS TABLE 
AS
RETURN 
(

SELECT
VE_Desc
,Dlr_dealership
/*
(case
when Dlr_dealership = 'Sonic Automotive-CMA' then [Act_Daily_Cash_Balance]-[Act_Daily_CDA]-[Act_Daily_Deposits] + 46438.39 -- 46495.39
--when Dlr_dealership = 'Tree Trunk, Inc' and ACT_DATE > '20140501'  then  [Act_Daily_Cash_Balance]-[Act_Daily_CDA]-[Act_Daily_Deposits] + 57
else [Act_Daily_Cash_Balance]-[Act_Daily_CDA]-[Act_Daily_Deposits] end) as BeginningBalance
--,[Act_Daily_Cash_Balance]-[Act_Daily_CDA]-[Act_Daily_Deposits] as BeginningBalance
*/
,[BeginingBalance]
,[Act_Daily_Deposits]
,[Act_Daily_CDA]
,[SweepAccountTotal] 
,(case
when Dlr_dealership = 'Sonic Automotive-CMA' then [Act_Daily_Cash_Balance] + 46438.39 --46495.39 
--when Dlr_dealership = 'Tree Trunk, Inc' and ACT_DATE > '20140501' then [Act_Daily_Cash_Balance]-[Act_Daily_CDA]-[Act_Daily_Deposits] + 57
else [Act_Daily_Cash_Balance] end) as [Act_Daily_Cash_Balance]
,[EndingBalance]

FROM  [cmag].[tblActivity]
join cmag.tblDealershipInfo
on Act_DLR_ID = Dlr_id
join  cmag.tblValidationElements
on Dlr_VE_Region_ID = VE_ID
Join cmag.vw_ActivityTbl_Security		ug
on 	[cmag].[tblActivity].Act_DLR_ID = ug.CMD_DLR_ID





where Act_Date = @ActivityDate
and Dlr_active = 1
--and DLR_ID <> '226'
--and DLR_ID <> '236'


)

CREATE FUNCTION [cmag].[tvfn_dearlerinfo] 
(@userid varchar(55)
)
RETURNS TABLE 
AS
RETURN 
(


select  top 100 percent dlr_dealership from CMS.cmag.tbldealershipinfo


join CMS.cmag.tblValidationElements
on Dlr_VE_Region_ID = VE_ID
--on  Dlr_id =  VE_VT_ID
Join [cmag].[vw_dealerinfoTbl_Security]		ug
on 	 Dlr_id = ug.CMD_DLR_ID

order by dlr_dealership


)
